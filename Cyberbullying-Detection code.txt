        // --- Page 4: Live Simulation ---
        const DemoPage = () => {
            const [tweet, setTweet] = useState('');
            const [result, setResult] = useState(null);
            const [isLoading, setIsLoading] = useState(false);

            const bullyingKeywords = ['stupid', 'idiot', 'hate', 'ugly', 'dumb', 'kill', 'loser', 'worthless', 'sex', 'fuck', 'dickhead', 'wtf', 'fat', 'die', 'shut up', 'bitch', 'asshole'];

            const analyzeTweet = useCallback(() => {
                if (tweet === '') return;

                setIsLoading(true);
                setResult(null);
                
                // Simulate backend processing delay
                setTimeout(() => {
                    const input = tweet.toLowerCase().trim();
                    let isBullying = false;
                    
                    bullyingKeywords.forEach(word => {
                        if (input.includes(word)) {
                            isBullying = true;
                        }
                    });

                    // Calculate confidence and status based on simulation logic
                    const confidence = isBullying 
                        ? (90 + Math.random() * 8).toFixed(1) // High confidence for bullying
                        : (90 + Math.random() * 9).toFixed(1); // High confidence for clean

                    setResult({
                        isBullying,
                        confidence: parseFloat(confidence),
                        status: isBullying ? 'CYBERBULLYING DETECTED' : 'CONTENT IS CLEAN',
                        desc: isBullying 
                            ? 'The MTF-IDF feature vectors strongly indicate offensive content.' 
                            : 'The model found no strong indicators of harmful language.'
                    });
                    setIsLoading(false);
                }, 1200);
            }, [tweet]);

            return (
                <div className="max-w-7xl mx-auto px-4 py-12 sm:px-6 lg:px-8">
                    <div className="bg-gradient-to-br from-stone-900 to-stone-800 rounded-2xl p-8 md:p-12 text-white shadow-xl">
                        <div className="max-w-3xl mx-auto text-center space-y-6">
                            <h2 className="text-3xl font-bold">Live Classification Simulation</h2>
                            <p className="text-stone-300">
                                Enter any text to see how the trained SVM model (simulated) classifies the input. This demonstrates the final application of the project pipeline.
                            </p>

                            <div className="bg-white/10 p-6 rounded-xl backdrop-blur-sm border border-white/10 mt-8">
                                <textarea 
                                    id="tweetInput" 
                                    rows="3" 
                                    className="w-full bg-stone-800/50 border border-stone-600 rounded-lg p-4 text-white placeholder-stone-500 focus:ring-2 focus:ring-indigo-500 focus:border-transparent outline-none transition-all"
                                    placeholder="Type a tweet here... (e.g., 'You are an idiot' or 'Have a nice day')"
                                    value={tweet}
                                    onChange={(e) => setTweet(e.target.value)}
                                />
                                
                                <div className="flex justify-end mt-4">
                                    <button 
                                        onClick={analyzeTweet}
                                        disabled={!tweet || isLoading}
                                        className={`px-6 py-2 rounded-lg font-medium transition-all shadow-lg flex items-center 
                                            ${!tweet || isLoading ? 'bg-indigo-400 cursor-not-allowed' : 'bg-indigo-600 hover:bg-indigo-500 hover:shadow-indigo-500/30'}`}
                                    >
                                        {isLoading ? 'Analyzing...' : 'Analyze Text'}
                                        <span className="ml-2 text-lg">➞</span>
                                    </button>
                                </div>
                            </div>

                            {/* Result Display */}
                            {result && (
                                <div className={`mt-8 transition-all duration-500 ${isLoading ? 'hidden' : ''}`}>
                                    <div className={`bg-white text-stone-900 p-6 rounded-xl shadow-lg border-l-8 ${result.isBullying ? 'border-red-600' : 'border-green-600'}`}>
                                        <h3 className="text-sm font-bold uppercase tracking-widest text-stone-500 mb-2">Classification Result</h3>
                                        <div className={`text-3xl font-black mb-2 ${result.isBullying ? 'text-red-600' : 'text-green-600'}`}>
                                            {result.status}
                                        </div>
                                        <p className="text-stone-600">{result.desc}</p>
                                        <div className="w-full bg-stone-200 h-2 rounded-full mt-4 overflow-hidden">
                                            <div 
                                                className={`h-full ${result.isBullying ? 'bg-red-600' : 'bg-green-600'} transition-all duration-1000`} 
                                                style={{ width: `${result.confidence}%` }}>
                                            </div>
                                        </div>
                                        <div className="flex justify-between text-xs text-stone-400 mt-1">
                                            <span>Confidence Score</span>
                                            <span className="font-bold text-stone-600">{result.confidence}%</span>
                                        </div>
                                    </div>
                                </div>
                            )}

                            {isLoading && (
                                <div className="mt-8">
                                    <div className="bg-white p-6 rounded-xl shadow-lg flex items-center justify-center space-x-3">
                                        <div className="w-4 h-4 border-2 border-indigo-600 border-t-transparent rounded-full animate-spin"></div>
                                        <span className="text-stone-800 font-medium">Model is classifying input...</span>
                                    </div>
                                </div>
                            )}

                        </div>
